use stockdb;
drop table hist_data;
create table hist_data(
	code char(6),
	date varchar(20),
	open double(20,2),
	high double(20,2),
	close double(20,2),
	low double(20,2),
	volume double(20,2),
	price_change  double(20,2),
	p_change  double(20,2),
	ma5 double(20,2),
	ma10 double(20,2),
	ma20 double(20,2),
	v_ma5 double(20,2),
	v_ma10 double(20,2),
	v_ma20 double(20,2),
	turnover  double(20,2)
)default charset=utf8;

drop table stock_basics;
create table stock_basics(
	code char(6),
	name varchar(20),
	industry varchar(20),
	area varchar(20),
	pe double(20,2),
	outstanding double(20,2),
	totals double(20,2),
	totalAssets double(20,2),
	primary key(code),
	unique key cd(code)
)default charset=utf8;
drop table profit_data;
create table profit_data(
       code char(6),
       name varchar(20),
       roe  double(20,3),
       net_profit_ratio double(20,3),
       gross_profit_rate double(20,3),
       net_profits double(20,3),
       eps double(20,3),
       business_income double(20,3),
       bips double(20,3)
)default charset=utf8;
